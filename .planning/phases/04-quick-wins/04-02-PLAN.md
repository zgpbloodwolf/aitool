```typescript
import { Tray, Menu, nativeImage, BrowserWindow, ipcMain, dialog, app } from 'electron'
import { join } from 'path'

let tray: Tray | null = null  // Must keep reference to prevent GC (Pitfall #2)

// Simple JSON file for closeBehavior — main process can't access localStorage
// This stores ONLY the close behavior preference (D-11 + D-06 compromise)
// All other settings use localStorage via renderer
import { readFileSync, writeFileSync, existsSync } from 'fs'

const CLOSE_BEHAVIOR_FILE = join(app.getPath('userData'), 'close-behavior.json')

type CloseBehavior = 'minimize' | 'quit' | 'ask'

function loadCloseBehavior(): CloseBehavior {
  try {
    if (existsSync(CLOSE_BEHAVIOR_FILE)) {
      const data = JSON.parse(readFileSync(CLOSE_BEHAVIOR_FILE, 'utf-8'))
      return data.behavior || 'ask'
    }
  } catch { /* ignore */ }
  return 'ask'
}

function saveCloseBehavior(behavior: CloseBehavior): void {
  writeFileSync(CLOSE_BEHAVIOR_FILE, JSON.stringify({ behavior }), 'utf-8')
}

export function setupTray(mainWindow: BrowserWindow): void {
  // Resolve icon path — works for both dev and production
  const iconPath = join(__dirname, '../../build/icon.ico')
  const icon = nativeImage.createFromPath(iconPath)

  tray = new Tray(icon)
  tray.setToolTip('AI 工具')

  const contextMenu = Menu.buildFromTemplate([
    {
      label: '新建对话',
      click: () => {
        mainWindow.show()
        mainWindow.focus()
        mainWindow.webContents.send('shortcut:new-tab')
      }
    },
    { type: 'separator' },
    {
      label: '显示窗口',
      click: () => {
        mainWindow.show()
        mainWindow.focus()
      }
    },
    { type: 'separator' },
    {
      label: '退出',
      click: () => {
        app.quit()
      }
    }
  ])

  tray.setContextMenu(contextMenu)

  // Double-click restores window
  tray.on('double-click', () => {
    mainWindow.show()
    mainWindow.focus()
  })

  // Close behavior management
  let isQuitting = false

  app.on('before-quit', () => {
    isQuitting = true
  })

  mainWindow.on('close', (e) => {
    if (isQuitting) return  // Allow quit when user chose Exit

    const behavior = loadCloseBehavior()

    if (behavior === 'ask') {
      // First time — ask user and remember choice
      e.preventDefault()
      dialog.showMessageBox(mainWindow, {
        type: 'question',
        title: '关闭确认',
        message: '关闭窗口时您希望执行什么操作？',
        buttons: ['最小化到托盘', '退出应用'],
        defaultId: 0,
        noLink: true
      }).then(({ response }) => {
        const chosen: CloseBehavior = response === 0 ? 'minimize' : 'quit'
        saveCloseBehavior(chosen)

        // Also sync to renderer's settings store via IPC
        mainWindow.webContents.send('settings:close-behavior-changed', chosen)

        if (chosen === 'minimize') {
          mainWindow.hide()
        } else {
          app.quit()
        }
      })
    } else if (behavior === 'minimize') {
      e.preventDefault()
      mainWindow.hide()
    }
    // behavior === 'quit' — allow default close (app.quit)
  })
}

// IPC handler for renderer to update close behavior from SettingsDrawer
export function registerTrayHandlers(): void {
  ipcMain.on('settings:update-close-behavior', (_event, behavior: CloseBehavior) => {
    saveCloseBehavior(behavior)
  })
}
```

Key implementation notes:
- `let tray: Tray | null = null` at module level prevents GC (Pitfall #2 from RESEARCH)
- Close behavior stored in `app.getPath('userData')/close-behavior.json` because main process cannot access localStorage (Pitfall #5)
- The settings drawer's tray radio group writes via IPC `'settings:update-close-behavior'` to sync the value
- `isQuitting` flag ensures tray's "Exit" and app.quit() from before-quit bypass the close handler
