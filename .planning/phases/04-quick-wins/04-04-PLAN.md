**Create `src/main/notification/notification-manager.ts`:**

Per D-01, D-02, D-04 implement:

```typescript
import { BrowserWindow, screen, ipcMain } from 'electron'
import { join } from 'path'

interface NotificationEntry {
  window: BrowserWindow
  id: string
  height: number
  timeout: ReturnType<typeof setTimeout>
}

interface ShowNotificationOptions {
  type: 'complete' | 'permission' | 'plan' | 'reply' | 'error'
  title: string
  body: string
  channelId?: string
  // Permission specific
  requestId?: string
  toolName?: string
}

const MAX_NOTIFICATIONS = 3
const NOTIFICATION_WIDTH = 360
const MARGIN = 10
const DEFAULT_TIMEOUT = 5000
const activeNotifications: NotificationEntry[] = []

export class NotificationManager {
  private mainWindow: BrowserWindow

  constructor(mainWindow: BrowserWindow) {
    this.mainWindow = mainWindow
    this.registerIpcHandlers()
  }

  private registerIpcHandlers(): void {
    // 通知窗口按钮点击处理
    ipcMain.on('notification:action', (_event, data: {
      notificationId: string
      action: string  // 'allow' | 'deny' | 'click' | 'view' | 'reply'
      replyText?: string
    }) => {
      const entry = activeNotifications.find(n => n.id === data.notificationId)
      if (!entry) return

      switch (data.action) {
        case 'allow':
        case 'deny':
          // 转发权限响应到 webview IPC（由 claude-webview.ts 处理）
          // Send permission response through the existing webview IPC
          this.mainWindow.webContents.send('notification:permission-response', {
            requestId: entry.window.webContents.metadata?.requestId,
            allowed: data.action === 'allow'
          })
          break
        case 'click':
        case 'view':
          // 跳转到对应标签页
          if (entry.window.webContents.metadata?.channelId) {
            this.mainWindow.show()
            this.mainWindow.focus()
            this.mainWindow.webContents.send('notification:focus-tab',
              entry.window.webContents.metadata.channelId)
          }
          break
      }
      this.dismiss(data.notificationId)
    })
  }

  show(options: ShowNotificationOptions): void {
    // 超过上限时移除最旧的通知
    while (activeNotifications.length >= MAX_NOTIFICATIONS) {
      this.dismiss(activeNotifications[0].id)
    }

    const { width: screenWidth, height: screenHeight } =
      screen.getPrimaryDisplay().workAreaSize

    // 计算堆叠位置
    const yOffset = activeNotifications.reduce(
      (acc, n) => acc + n.height + MARGIN, 0
    )

    const notifHeight = options.type === 'permission' ? 160
      : options.type === 'reply' ? 160
      : options.type === 'error' ? 120
      : 100

    const x = screenWidth - NOTIFICATION_WIDTH - MARGIN
    const y = screenHeight - notifHeight - MARGIN - yOffset

    const id = `notif_${Date.now()}_${Math.random().toString(36).slice(2)}`

    const win = new BrowserWindow({
      width: NOTIFICATION_WIDTH,
      height: notifHeight,
      x, y,
      frame: false,
      transparent: true,
      alwaysOnTop: true,
      skipTaskbar: true,
      resizable: false,
      focusable: false,
      show: false,
      webPreferences: {
        preload: join(__dirname, '../preload/index.js'),  // 复用主 preload（通知窗口 IPC 有限）
        contextIsolation: true,
        sandbox: false
      }
    })

    // Store metadata for IPC routing
    win.webContents.metadata = {
      notificationId: id,
      channelId: options.channelId,
      requestId: options.requestId
    }

    // Load notification HTML with query parameters
    const params = new URLSearchParams({
      id,
      type: options.type,
      title: options.title,
      body: options.body,
      toolName: options.toolName || ''
    })

    // For dev mode, notification HTML needs to be loadable
    if (!require('electron').app.isPackaged) {
      // Dev: load from src/notification/index.html directly
      win.loadFile(join(__dirname, '../../src/notification/index.html'), { query: Object.fromEntries(params) })
    } else {
      // Production: load from out directory
      win.loadFile(join(__dirname, '../notification/index.html'), { query: Object.fromEntries(params) })
    }

    // Show with slide-in animation (handled by CSS)
    win.once('ready-to-show', () => {
      win.show()
      // Reposition existing notifications
      this.repositionAll()
    })

    const timeout = setTimeout(() => {
      this.dismiss(id)
    }, DEFAULT_TIMEOUT)

    activeNotifications.push({ window: win, id, height: notifHeight, timeout })
  }

  dismiss(id: string): void {
    const idx = activeNotifications.findIndex(n => n.id === id)
    if (idx === -1) return
    const entry = activeNotifications[idx]
    clearTimeout(entry.timeout)
    entry.window.close()
    activeNotifications.splice(idx, 1)
    this.repositionAll()
  }

  private repositionAll(): void {
    const { width: screenWidth, height: screenHeight } =
      screen.getPrimaryDisplay().workAreaSize

    let yOffset = 0
    for (const entry of activeNotifications) {
      const x = screenWidth - NOTIFICATION_WIDTH - MARGIN
      const y = screenHeight - entry.height - MARGIN - yOffset
      entry.window.setBounds({ x, y, width: NOTIFICATION_WIDTH, height: entry.height })
      yOffset += entry.height + MARGIN
    }
  }

  dismissAll(): void {
    for (const entry of activeNotifications) {
      clearTimeout(entry.timeout)
      entry.window.close()
    }
    activeNotifications.length = 0
  }
}
```

  const settingsStore = useSettingsStore()

  // 监听通知事件（主进程检测到事件后发送给渲染进程判断可见性）
  window.api.onNotificationEvent((data) => {
    // D-03: 窗口在前台且对应标签页可见时不发通知
    if (document.hasFocus()) {
      // 检查是否是对应标签页 — 由 App.vue 的 ChatPanel ref 判断
      // 简化：如果窗口在前台，不显示桌面通知
      // 但仍然播放声音
      playSoundForType(data.type)
      return
    }

    // 检查设置中的通知开关
    if (!settingsStore.settings.notifyEnabled) return
    const toggleMap: Record<string, boolean> = {
      complete: settingsStore.settings.notifyComplete,
      permission: settingsStore.settings.notifyPermission,
      plan: settingsStore.settings.notifyPlan,
      reply: settingsStore.settings.notifyReply,
      error: settingsStore.settings.notifyError
    }
    if (!toggleMap[data.type]) return

    // 通知主进程显示通知窗口
    window.api.showNotification(data)

    // 播放声音
    playSoundForType(data.type)
  })

  // 处理通知窗口的跳转请求
  window.api.onNotificationFocusTab?.((channelId: string) => {
    // 由 App.vue / ChatPanel 处理标签页切换
    // 发送到 ChatPanel
  })
}

function playSoundForType(type: string): void {
  if (type === 'complete') playSound('complete')
  else if (type === 'permission') playSound('permission')
  else if (type === 'error') playSound('error')
}
