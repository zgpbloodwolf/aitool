      2. 销毁 ghost 窗口

      6. 清除 drag 状态
      7. 返回新窗口信息

    - `cancelDrag(): void` — 取消拖拽：停止 interval，销毁 ghost，清除 drag 状态

    - `getDragState(): { channelId: string | null; tabId: string | null }` — 返回当前拖拽状态

    ```typescript
    import { BrowserWindow } from 'electron'

    // 标签拖拽出窗口 — 开始拖拽预览
    ipcMain.on('tab-drag:start', (event, data: { channelId: string; tabId: string }) => {
      windowManager.startDragPreview(data.channelId, data.tabId)
    })

    // 标签拖拽出窗口 — 结束拖拽，创建新窗口
    ipcMain.handle('tab-drag:end', (event) => {
      // 从 event.sender 获取源窗口 ID
      const sourceWindow = BrowserWindow.fromWebContents(event.sender)
      const sourceWindowId = sourceWindow?.id
      if (!sourceWindowId) return { success: false, error: '无法确定源窗口' }

      const result = windowManager.finalizeDragOut(sourceWindowId)
      if (result) {
        // 通知新窗口恢复标签页（解决 "新窗口打开空标签列表" 问题）
        const dragState = windowManager.getDragState()
        if (dragState.channelId) {
          const sourceWin = BrowserWindow.fromWebContents(event.sender)
          const sourceCwd = sourceWin ? await getChannelCwd(dragState.channelId) : undefined
          result.window.webContents.send('window:restore-tab', {
            channelId: dragState.channelId,
            tabId: dragState.tabId,
            label: dragState.tabId || '对话',
            cwd: sourceCwd
          })
        }
        return { success: true, windowId: result.windowId, channelId: dragState.channelId, tabId: dragState.tabId }
      }
      return { success: false, error: '拖拽取消或失败' }
    })

    // 标签拖拽出窗口 — 取消拖拽
    ipcMain.on('tab-drag:cancel', () => {
      windowManager.cancelDrag()
    })
    ```

    ```typescript
    /** 检测鼠标是否接近窗口边缘 — 触发拖出行为 */
    function checkDragOutWindow(e: DragEvent): void {
      if (!draggedTabId.value || dragOutNotified) return

      const margin = dragOutThreshold
      const nearEdge = e.clientX < margin || e.clientX > window.innerWidth - margin ||
                       e.clientY < margin

      if (nearEdge) {
        // 获取拖拽标签对应的 channelId
        let channelId: string | null = null
        for (const [chId, tabId] of channelToTab) {
          if (tabId === draggedTabId.value) {
            channelId = chId
            break
          }
        }
        if (channelId) {
          isDraggingOut.value = true
          dragOutNotified = true
          window.api.tabDragStart({ channelId, tabId: draggedTabId.value })
        }
      }
    }
    ```

    ```typescript
    /** D-16/UX-11: 拖拽结束 — 排序完成或拖出完成 */
    async function onDragEnd(): Promise<void> {
      if (isDraggingOut.value && draggedTabId.value) {
        // 拖拽出窗口 — 通知主进程创建新窗口
        const result = await window.api.tabDragEnd()
        if (result.success) {
          // 新窗口需要恢复标签 — 发送恢复指令到新窗口
          // 主进程已通过 window:restore-tab IPC 发送到新窗口
          // 关闭原窗口的标签和频道
          const tabId = draggedTabId.value
          let channelId: string | null = null
          for (const [chId, tId] of channelToTab) {
            if (tId === tabId) { channelId = chId; break }
          }
          if (channelId) {
            // 不发送 close_channel（频道已迁移到新窗口），只清理本地状态
            channelToTab.delete(channelId)
          }
          // 从标签列表移除
          const idx = tabs.value.findIndex(t => t.id === tabId)
          if (idx !== -1) tabs.value.splice(idx, 1)
          iframeRefs.value.delete(tabId)
          if (tabs.value.length === 0) addNewTab()
          else if (activeTabId.value === tabId) {
            activeTabId.value = tabs.value[Math.min(idx, tabs.value.length - 1)].id
          }
        } else {
          // 拖拽失败 — 取消
          window.api.tabDragCancel()
        }
      } else if (dragOutNotified) {
        // 鼠标回到窗口内 — 取消拖出
        window.api.tabDragCancel()
      }

      // 清除拖拽状态
      draggedTabId.value = null
      dragOverTabId.value = null
      isDraggingOut.value = false
      dragOutNotified = false
    }
    ```
